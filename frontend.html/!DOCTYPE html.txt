<script>
const API_URL = "http://127.0.0.1:5000/chat"; // replace later with Render URL
function addMessage(sender, text) {
 const messages = document.getElementById('messages');
 const div = document.createElement('div');
 div.className = sender;
 div.textContent = `${sender === 'user' ? 'You' : 'Tera'}: ${text}`;
 messages.appendChild(div);
 messages.scrollTop = messages.scrollHeight;
}
function sendMessage() {
 const input = document.getElementById('userInput');
 const message = input.value.trim();
 if (!message) return;
 addMessage('user', message);
 input.value = '';
 fetch(API_URL, {
   method: 'POST',
   headers: {'Content-Type': 'application/json'},
   body: JSON.stringify({message})
 })
 .then(res => res.json())
 .then(data => addMessage('bot', data.response))
 .catch(() => addMessage('bot', "Oops! You made me short circuit ðŸ˜…"));
}
</script>
